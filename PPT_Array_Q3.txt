    public int ValidateVal(int [] sortedArray, int searchVal) 
    {
        int res=0;
        for(int i=0;i<sortedArray.length;i++){
            
                if(sortedArray[i]==searchVal) {
                    res=i;
                }
        }
        
        if(res==0)
        {
            outter:
            for(int i=0;i<sortedArray.length;i++){
                if(sortedArray[i]<searchVal){
                    i++;
                } else {
                    res=i;
                    break outter;
                }
            }
        }
        
        return res+1;
    }
    
	public static void main(String[] args) {
